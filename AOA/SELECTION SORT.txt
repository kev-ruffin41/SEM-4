SELECTION SORT

#include<stdio.h>
#include<stdlib.h>
void main()
{
	int i,j,n,temp,flag=0;
	int a[100];
	printf("Enter number of elements\n");
	scanf("%d",&n);
	printf("Enter the elements\n");
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			if(a[i]>a[j])
			{
				temp=a[j];
				a[j]=a[i];
				a[i]=temp;
			}
			
		}
	}
	printf("Sorted elements are\n");
	for(i=0;i<n;i++)
	{
		printf("%d\n",a[i]);
	}
}
