LINEAR SEARCH AND  BINARY SEARCH

#include<stdio.h>
void Lsearch(int a[],int r,int n)
{   
	int i,flag=0;
  	for(i=0;i<n;i++)
	{
      
	  if(a[i]==r)
       	 {
           		flag=1;
           		printf("Element %d found at %d location.",r,i+1);
          		break;
    	 }
    }
    if(flag==0)
    {
        printf("Element not found.");
    }
}


int Bsearch(int a[],int low,int high,int r)
{   int flag=0;
    while(low<high)
    {
        int mid=(low+high)/2;
        if(a[mid]==r)
            return mid;
        else if(a[mid]<r)
            Bsearch(a,mid+1,high,r);
        else
            Bsearch(a,low,mid,r);
    }
    if(flag==0)
        return -1;
}
